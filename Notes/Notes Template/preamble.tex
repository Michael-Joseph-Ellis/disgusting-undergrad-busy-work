\newcommand*\circled[1]
{
	\tikz[baseline = (char.base)]
	{
		\node[shape = circle, draw, inner sep = 1pt] (char) {#1};
	}

{
	\ifnumequal{\value{#1}}{0}
	{??}
	{\the\value{#1}}
